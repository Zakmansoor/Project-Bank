    int& secondNumber = firstNumber;

    cout << firstNumber << endl;//10
    cout << secondNumber << endl;//10

    secondNumber = 20;
    cout << firstNumber << endl;//20
    cout << secondNumber << endl;//20

    firstNumber = 30;
    cout << firstNumber << endl;//30
    cout << secondNumber;//30
}
